% Kurzaufgabe: \kurzaufgabe[spalten]{Aufgabentext}{Teilaufgaben als \item}
\newcommand{\kurzaufgabe}[3][2]{%
	\refstepcounter{exo}%
	\refstepcounter{exoglobal}%
	\item \label{aufg:\aufgabenlabel}%
	#2% Aufgabentext
	\ifx&#3&% Wenn keine Teilaufgaben (Parameter 3 leer)
		% Keine Teilaufgaben
	\else
		\par\noindent
		\begin{multicols}{#1}
		\begin{enumerate}[label=\alph*)]
			#3% Teilaufgaben
		\end{enumerate}
		\end{multicols}
	\fi
	\par\noindent\hfill\zumloesung{\aufgabenlabel}%
}
